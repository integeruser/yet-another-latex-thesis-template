\newcommand\sourcelink[4]{%
    \ifstrempty{#4}{%
        \ifstrempty{#3}{%
            \href{#1#2}{\path{#2}}%
        }{%
            \href{#1#2\#L#3}{\path{#2##L#3}}%
        }%
    }{%
        \href{#1#2\#L#3}{\path{#2##L#3}}/\href{#1#2\#L#4}{\path{##L#4}}%
    }%
}
